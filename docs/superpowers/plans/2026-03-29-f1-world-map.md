## Design Brief

- **Aesthetic:** JRPG inventory screen — sharp pixel corners, pixel drop-shadows, sprite-like UI panels, subtle scanline texture
- **Fonts:** Press Start 2P (headings/labels), Geist Sans (body text — already installed)
- **Palette:**
  - Background: `#0A0F14` (near-black)
  - Accent: `#0D9488` (teal-600)
  - Visited fill: `#0F766E` (teal-700), opacity 0.8
  - Planned fill: `#B45309` (amber-700), opacity 0.5
  - Wishlist: no fill, `#6D28D9` (violet-700) dashed stroke
  - Unvisited: transparent (map tiles show through)
- **Map tiles:** CartoDB Dark Matter (dark background, subtle labels)
- **UI panels:** `bg-slate-900/90` backdrop-blur, 2px teal border, 4px offset pixel shadow

## Brainstorming Notes & Key Decisions

### Country Status Logic
Countries don't have their own "status" field — status is derived from linked Trips:
- **Visited** → any linked Trip with `Status = "Completed"`
- **Planned** → no Completed, but has `Status = "Planned"` or `"In Progress"`
- **Wishlist** → only `Status = "Wishlist"` trips
- **Unvisited** → not linked to any trip

Fetch Countries + Cities with `revalidate: 3600` (change rarely). Fetch Trips with `cache: 'no-store'` (need current status). Join server-side and pass to the client as a pre-computed map.

### GeoJSON Matching
Natural Earth 50m GeoJSON uses `ISO_A2` property (2-letter codes) to match Airtable's `ISO Code` field. Some territories have `"-99"` — skip those in the CountryLayer render.

### Stripe/Hatch Pattern for Planned Countries
SVG `<pattern>` injection into Leaflet's overlay SVG element. On mount, insert a `<defs>` block with a diagonal stripe pattern and reference it via `fillColor: "url(#planned-stripe)"`. Falls back to amber semi-transparent if the SVG isn't available yet.

### Full-Screen Map Layout
Dashboard layout is transparent — children fill the viewport. Nav is a floating pill anchored top-center (absolute positioned over the map). Country sidebar slides in from the right without pushing the map. On mobile: sidebar becomes a bottom sheet (fixed, 60vh, slide-up animation).

### Leaflet in Next.js App Router
Leaflet requires DOM — must be dynamically imported with `ssr: false`. Pattern:
```tsx
// page.tsx (server component)
const WorldMap = dynamic(() => import('@/components/map/WorldMap'), { ssr: false })
```
Leaflet CSS must be imported inside the client component, not at the module level (avoids SSR crash).

### Task 5: Map Page (Server Component)

**Files:** `app/(dashboard)/map/page.tsx`

This is the data-fetching layer. It runs on the server, calls the Airtable API routes, builds the country status map, and passes clean typed data to the WorldMap client component.

- [ ] **5.1 — Create the map page**

Create `app/(dashboard)/map/page.tsx`:

```tsx
import dynamic from 'next/dynamic'
import { listRecords } from '@/lib/airtable'
import type { CountryFields, CityFields, TripFields } from '@/lib/types/airtable'
import type { CountryMapData, CityMapData, TripSummary, CountryStatus } from '@/lib/types/map'

// Dynamic import — Leaflet cannot run on the server
const WorldMap = dynamic(() => import('@/components/map/WorldMap'), {
  ssr: false,
  loading: () => (
    <div className="flex h-screen w-screen items-center justify-center bg-[#0A0F14]">
      <p
        className="text-teal-400 text-[10px] animate-pulse"
        style={{ fontFamily: 'var(--font-pixel)' }}
      >
        LOADING MAP...
      </p>
    </div>
  ),
})

// Helper: derive country status from its linked trips
function deriveCountryStatus(
  countryId: string,
  trips: Array<{ id: string } & TripFields>
): CountryStatus {
  const linked = trips.filter((t) => t.Country?.includes(countryId))
  if (linked.some((t) => t.Status === 'Completed')) return 'visited'
  if (linked.some((t) => t.Status === 'Planned' || t.Status === 'In Progress'))
    return 'planned'
  if (linked.some((t) => t.Status === 'Wishlist')) return 'wishlist'
  return 'unvisited'
}

export default async function MapPage() {
  // Fetch in parallel — countries + cities are slow-changing, trips need fresh status
  const [rawCountries, rawCities, rawTrips] = await Promise.all([
    listRecords('Countries'),   // server-side, Next.js ISR revalidates at 3600s
    listRecords('Cities'),
    listRecords('Trips'),
  ])

  // Cast raw records to typed shapes
  const countries = rawCountries as Array<{ id: string } & CountryFields>
  const cities = rawCities as Array<{ id: string } & CityFields>
  const trips = rawTrips as Array<{ id: string } & TripFields>

  // Build TripSummary lookup by trip ID
  const tripSummaryMap = new Map<string, TripSummary>(
    trips.map((t) => [
      t.id,
      {
        id: t.id,
        name: t.Name,
        status: t.Status,
        startDate: t['Start Date'],
        endDate: t['End Date'],
        rating: t.Rating,
      },
    ])
  )

  // Build CountryMapData
  const countryData: CountryMapData[] = countries
    .filter((c) => c['ISO Code'])
    .map((c) => ({
      id: c.id,
      name: c.Name,
      isoCode: c['ISO Code'],
      flagEmoji: c['Flag Emoji'],
      continent: c.Continent,
      status: deriveCountryStatus(c.id, trips),
      timesVisited: c['Times Visited'] ?? 0,
      firstVisited: c['First Visited'],
      trips: (c.Trips ?? [])
        .map((tid) => tripSummaryMap.get(tid))
        .filter((t): t is TripSummary => t !== undefined),
    }))

  // Build CityMapData — only cities with coordinates
  const cityData: CityMapData[] = cities
    .filter((c) => c.Latitude != null && c.Longitude != null)
    .map((c) => ({
      id: c.id,
      name: c.Name,
      latitude: c.Latitude!,
      longitude: c.Longitude!,
      rating: c.Rating,
      timesVisited: c['Times Visited'] ?? 0,
      trips: (c.Trips ?? [])
        .map((tid) => tripSummaryMap.get(tid))
        .filter((t): t is TripSummary => t !== undefined),
    }))

  return <WorldMap countries={countryData} cities={cityData} />
}
```

**Cache configuration:** The `listRecords` calls go through the Airtable SDK which doesn't use native fetch, so Next.js ISR revalidation doesn't apply automatically. The data is fetched fresh on each server request in development. For production, wrap the page with `export const revalidate = 3600` for countries/cities. Since trips need to be fresh, use `export const dynamic = 'force-dynamic'` for now — the small data size makes this acceptable.

Add `export const dynamic = 'force-dynamic'` near the top of the file.

### Task 7: CountryLayer Component

**Files:** `components/map/CountryLayer.tsx`

Renders the GeoJSON country boundaries with status-based fill colours. Injects an SVG diagonal stripe `<pattern>` for Planned countries.

- [ ] **7.1 — Create CountryLayer.tsx**

Create `components/map/CountryLayer.tsx`:

```tsx
'use client'

import { useEffect, useRef } from 'react'
import { GeoJSON, useMap } from 'react-leaflet'
import type { GeoJsonObject, Feature } from 'geojson'
import type { PathOptions, LeafletMouseEvent } from 'leaflet'
import type { CountryMapData, CountryStatus } from '@/lib/types/map'
import countriesGeoJson from '@/public/data/countries.geojson'

// Status → Leaflet path options
function getPathOptions(status: CountryStatus): PathOptions {
  switch (status) {
    case 'visited':
      return {
        fillColor: '#0F766E',
        fillOpacity: 0.78,
        color: '#0D9488',
        weight: 1,
      }
    case 'planned':
      return {
        fillColor: 'url(#planned-stripe)',
        fillOpacity: 1,
        color: '#D97706',
        weight: 1.5,
      }
    case 'wishlist':
      return {
        fillColor: 'transparent',
        fillOpacity: 0,
        color: '#7C3AED',
        weight: 2,
        dashArray: '6 4',
      }
    default: // unvisited
      return {
        fillColor: '#0F172A',
        fillOpacity: 0.15,
        color: '#1E293B',
        weight: 0.5,
      }
  }
}

interface CountryLayerProps {
  countries: CountryMapData[]
  onCountryClick: (country: CountryMapData) => void
}

export default function CountryLayer({ countries, onCountryClick }: CountryLayerProps) {
  const map = useMap()
  const patternInjected = useRef(false)

  // Build a lookup: ISO code → country data
  const countryByIso = new Map<string, CountryMapData>(
    countries.map((c) => [c.isoCode, c])
  )

  // Inject SVG stripe pattern into Leaflet's overlay SVG for Planned countries
  useEffect(() => {
    if (patternInjected.current) return
    const svgEl = map.getPanes().overlayPane?.querySelector('svg')
    if (!svgEl) return
    if (svgEl.querySelector('#planned-stripe')) return

    const defs = document.createElementNS('http://www.w3.org/2000/svg', 'defs')
    defs.innerHTML = `
      <pattern id="planned-stripe" patternUnits="userSpaceOnUse" width="8" height="8" patternTransform="rotate(45 0 0)">
        <rect width="8" height="8" fill="#78350F" fill-opacity="0.45"/>
        <line x1="0" y1="0" x2="0" y2="8" stroke="#D97706" stroke-width="3" stroke-opacity="0.7"/>
      </pattern>
    `
    svgEl.insertBefore(defs, svgEl.firstChild)
    patternInjected.current = true
  }, [map])

  function style(feature: Feature | undefined): PathOptions {
    const iso = feature?.properties?.ISO_A2 as string | undefined
    if (!iso || iso === '-99') return getPathOptions('unvisited')
    const country = countryByIso.get(iso)
    return getPathOptions(country?.status ?? 'unvisited')
  }

  function onEachFeature(feature: Feature, layer: L.Layer) {
    const iso = feature.properties?.ISO_A2 as string | undefined
    if (!iso || iso === '-99') return

    const country = countryByIso.get(iso)
    if (!country || country.status === 'unvisited') return

    layer.on({
      click: () => onCountryClick(country),
      mouseover: (e: LeafletMouseEvent) => {
        ;(e.target as L.Path).setStyle({ weight: 2, fillOpacity: 0.9 })
      },
      mouseout: (e: LeafletMouseEvent) => {
        ;(e.target as L.Path).setStyle(style(feature))
      },
    })
  }

  return (
    <GeoJSON
      key={countries.length} // re-render if data changes
      data={countriesGeoJson as GeoJsonObject}
      style={style}
      onEachFeature={onEachFeature}
    />
  )
}
```

**Note on GeoJSON import:** Importing the JSON file directly requires `"resolveJsonModule": true` in tsconfig.json (already present in most Next.js projects). If TypeScript errors on the import, fetch it instead:
```tsx
// Alternative fetch approach (if JSON import fails):

useEffect(() => {
  fetch('/data/countries.geojson').then(r => r.json()).then(setGeoData)
}, [])
if (!geoData) return null
```

Prefer the static import for performance; use the fetch fallback if TypeScript objects.

'use client'

import L from 'leaflet'
import { Marker, Popup } from 'react-leaflet'
import type { CityMapData } from '@/lib/types/map'

function createCityIcon(timesVisited: number): L.DivIcon {
  const size = timesVisited > 3 ? 12 : timesVisited > 1 ? 10 : 8
  return L.divIcon({
    html: `
      <div style="
        width: ${size}px;
        height: ${size}px;
        background: #0D9488;
        border: 2px solid #5EEAD4;
        box-shadow: 0 0 ${size}px #0D9488, 2px 2px 0 #0F766E;
        cursor: pointer;
      "></div>
    `,
    className: '',
    iconSize: [size, size],
    iconAnchor: [size / 2, size / 2],
  })
}

interface CityMarkersProps {
  cities: CityMapData[]
}

export default function CityMarkers({ cities }: CityMarkersProps) {
  return (
    <>
      {cities.map((city) => (
        <Marker
          key={city.id}
          position={[city.latitude, city.longitude]}
          icon={createCityIcon(city.timesVisited)}
        >
          <Popup>
            <div className="min-w-[160px]">
              <p
                className="text-[9px] text-teal-300 mb-2"
                style={{ fontFamily: 'var(--font-pixel)' }}
              >
                {city.name.toUpperCase()}
              </p>
              {city.rating && (
                <p className="text-[11px] text-amber-400 mb-1">
                  {'★'.repeat(city.rating)}{'☆'.repeat(5 - city.rating)}
                </p>
              )}

'use client'

import { useEffect } from 'react'
import type { CountryMapData, TripSummary } from '@/lib/types/map'
import type { TripStatus } from '@/lib/types/airtable'

const STATUS_LABEL: Record<TripStatus, string> = {
  Completed: 'DONE',
  Planned: 'PLANNED',
  'In Progress': 'IN PROG',
  Wishlist: 'WISHLIST',
}

const STATUS_COLOR: Record<TripStatus, string> = {
  Completed: 'text-teal-400 border-teal-700',
  Planned: 'text-amber-400 border-amber-700',
  'In Progress': 'text-blue-400 border-blue-700',
  Wishlist: 'text-violet-400 border-violet-700',
}

interface CountrySidebarProps {
  country: CountryMapData | null
  onClose: () => void
}

export default function CountrySidebar({ country, onClose }: CountrySidebarProps) {
  // Close on Escape key
  useEffect(() => {
    const handler = (e: KeyboardEvent) => {
      if (e.key === 'Escape') onClose()
    }
    window.addEventListener('keydown', handler)
    return () => window.removeEventListener('keydown', handler)
  }, [onClose])

  const isOpen = country !== null

  return (
    <>
      {/* Desktop: slide-in from right */}
      <aside
        className={`
          hidden sm:flex flex-col
          fixed top-0 right-0 h-screen w-80 z-[999]
          pixel-panel border-l-0
          transition-transform duration-300 ease-out
          ${isOpen ? 'translate-x-0' : 'translate-x-full'}
        `}
      >
        <SidebarContent country={country} onClose={onClose} />
      </aside>

      {/* Mobile: bottom sheet */}
      <div
        className={`
          sm:hidden
          fixed bottom-0 left-0 right-0 z-[999] h-[60vh]
          pixel-panel border-b-0
          transition-transform duration-300 ease-out
          ${isOpen ? 'translate-y-0' : 'translate-y-full'}
        `}
      >
        <div className="flex justify-center pt-2 pb-1">
          <div className="w-12 h-1 bg-teal-700" />
        </div>
        <SidebarContent country={country} onClose={onClose} />
      </div>

      {/* Backdrop */}
      {isOpen && (
        <div
          className="fixed inset-0 z-[998] bg-black/30 sm:hidden"
          onClick={onClose}
        />
      )}
    </>
  )
}

function SidebarContent({
  country,
  onClose,
}: {
  country: CountryMapData | null
  onClose: () => void
}) {
  if (!country) return null

  return (
    <div className="flex flex-col h-full overflow-hidden">
      {/* Header */}
      <div className="flex items-start justify-between p-4 border-b border-teal-900">
        <div>
          <p className="text-2xl mb-1">{country.flagEmoji}</p>
          <h2
            className="text-[11px] text-teal-300 leading-relaxed"
            style={{ fontFamily: 'var(--font-pixel)' }}
          >
            {country.name.toUpperCase()}
          </h2>
          {country.continent && (
            <p className="text-[9px] text-slate-500 mt-0.5"
              style={{ fontFamily: 'var(--font-pixel)' }}>
              {country.continent.toUpperCase()}
            </p>
          )}
        </div>
        <button
          onClick={onClose}
          className="text-slate-500 hover:text-teal-300 text-[10px] mt-1 border border-slate-700 hover:border-teal-600 px-2 py-1 transition-colors"
          style={{ fontFamily: 'var(--font-pixel)' }}
          aria-label="Close"
        >
          ✕
        </button>
      </div>

      {/* Stats row */}
      <div className="flex gap-4 px-4 py-3 border-b border-teal-900/50">
        <Stat label="VISITS" value={String(country.timesVisited)} />
        {country.firstVisited && (
          <Stat label="FIRST" value={country.firstVisited.slice(0, 4)} />
        )}
        <Stat
          label="STATUS"
          value={country.status.toUpperCase().slice(0, 7)}
          valueClass={
            country.status === 'visited' ? 'text-teal-400' :
            country.status === 'planned' ? 'text-amber-400' :
            'text-violet-400'
          }
        />
      </div>

      {/* Trip list */}
      <div className="flex-1 overflow-y-auto px-4 py-3">
        {country.trips.length === 0 ? (
          <p className="text-[9px] text-slate-600 mt-2"
            style={{ fontFamily: 'var(--font-pixel)' }}>
            NO TRIPS YET
          </p>
        ) : (
          <ul className="space-y-2">
            {country.trips.map((trip) => (
              <TripRow key={trip.id} trip={trip} />
            ))}
          </ul>
        )}
      </div>
    </div>
  )
}

function Stat({ label, value, valueClass = 'text-teal-300' }: {
  label: string
  value: string
  valueClass?: string
}) {
  return (
    <div>
      <p className="text-[8px] text-slate-500" style={{ fontFamily: 'var(--font-pixel)' }}>
        {label}
      </p>
      <p className={`text-[10px] mt-0.5 ${valueClass}`}
        style={{ fontFamily: 'var(--font-pixel)' }}>
        {value}
      </p>
    </div>
  )
}

function TripRow({ trip }: { trip: TripSummary }) {
  return (
    <li className="border border-slate-800 hover:border-teal-900 p-2 transition-colors">
      <div className="flex items-center justify-between gap-2">
        <p className="text-[10px] text-slate-200 truncate flex-1">{trip.name}</p>
        <span
          className={`text-[8px] px-1.5 py-0.5 border shrink-0 ${STATUS_COLOR[trip.status]}`}
          style={{ fontFamily: 'var(--font-pixel)' }}
        >
          {STATUS_LABEL[trip.status]}
        </span>
      </div>
      {trip.startDate && (
        <p className="text-[9px] text-slate-500 mt-0.5">
          {trip.startDate.slice(0, 4)}
          {trip.endDate && trip.endDate.slice(0, 4) !== trip.startDate.slice(0, 4)
            ? `–${trip.endDate.slice(0, 4)}`
            : ''}
        </p>
      )}
    </li>
  )
}
